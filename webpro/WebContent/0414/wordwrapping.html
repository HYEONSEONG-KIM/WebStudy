ecma Script : 자바스크립트의 버전을 나타내는 표준 정의어 es2, es4, es6, es8
script에서 html요소에 접근
id를 이용 => 
doucument.getElementById('아이디이름') 
doucument.querySelector('#아이디이름') -> es6

태그이름을 이용 =>
doucument.getElementsByTagName('태그이름') -> 태그가 여러개의 경우 배열형태로 저장
a= doucument.getElementsByTagName('태그이름') 

document.querySelector('태그이름') -> 첫번째 태그만 선택
document.querySelectorAll('태그이름') -> 배열형태 

class를 이용
document.getElementsByClassName('class이름')[] -> 배열형태
doucument.querySelector('.class이름') -> 첫번째 요소만 접근
doucument.querySelectorAll('.class이름')

//div에 추가 하기 위해서 div요소에 접근
var a = document.getElementsByTagName('div')[0];

//일반 문자를 객체화 - Text Object로 변환
text = document.createTextNode(...);

//div에 자식객체로 추가
a.appendChild(text);
